server {
        listen 80 default_server;
        listen [::]:80 default_server;
        root /var/www/html;
        index index.html index.htm index.nginx-debian.html;
        server_name _;
        location / {
                try_files $uri $uri/ =404;
        }
}

server {
        listen 2657 camazon;
        listen [::]:2657 camazon;

server {  
    listen 2657;
    server_name localhost;

    root /var/www/html;
    index camazon.html;

    location / {
        try_files $uri $uri/ /index.html;
    }
